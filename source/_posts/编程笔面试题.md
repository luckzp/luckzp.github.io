### 1.字符串复制函数(strcpy)

```c++
char* strcpy(char* dst, const char* src)   
{
	assert(dst != NULL && src != NULL);    
	char *ret = dst;  
	while ((*dst++ = *src++) != '\0'); 
	return ret;
}
```

### 2.内存复制函数(memcpy)

```c++
void* memcpy2(void* dest, const void* src, int count)
{
	char* d;
	const char* s;
	if (dest >= ((char *)src + count) || dest <= src)
	{
		d = (char *)dest;
		s = (char *)src;
		while (count--)
			*d++ = *s++;
	}
	else //memory overlap
	{
		d = (char*)dest + count - 1;
		s = (char*)src + count - 1;
		while (count--)
			*d-- = *s--;
	}
	return dest;
}
```

### 3.合并两个有序的链表

```c++
Node* listMerge(Node* head1, Node* head2)
{
	if(head1 == NULL)
  		return head2;
  	if(head2 == NULL)
  		return head1
  	Node* head = NULL;
  	if(head1.val > head2.val)
  	{
      head = head2;
      head->next = listMerge(head1, head2->next);
  	}else
  	{
      head = head1;
      head->next = listMerge(head1->next, head2);
  	}
  	return head;
}
```

### 4.逆转链表

```c++
Node* reverseList(Node* head)
{
	Node* pre = NULL;
	Node* res = NULL;
  	while(head != NULL)
  	{
  		res = head->next;
  		head->next = pre;
  		pre = head;
  		head = res;
  	}
  return pre;
}
```
